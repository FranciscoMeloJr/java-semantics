//Octal and unicode escapes returning different results than JDK.
public class literals_90_char_unsupported {
	public static void main(String[] args) {
    System.out.println("Octal escape:");
    System.out.println(""+'\0'+'\7'+'\00'+'\07'+'\37'+'\77'
        +'\000'+'\007'+'\077'+'\177'+'\377');
    System.out.println("Unicode escape:");
    System.out.println(""+'\u0000'+'\uu0001'+'\uu0007'+'\uuu0010'+'\uuu007f'
        +'\u0080'+'\u00ff');
		System.out.println("Done!");
	}
}
